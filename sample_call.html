## 동작 규칙

1. 내 메시지에서 아래 keyword를 감지하세요:
   - [1박2일, 근교여행, 숙박] → travel_overnight.md
   - [당일치기, 산책, 나들이] → travel_daytrip.md
   - [맛집, 밥집, 카페, 뭐먹지] → food_nearby.md
   - [신규개발, 기능추가, 구현] → dev_newproject.md
   - [주식, 종목, 매수, 투자] → stock_analysis.md

2. keyword 감지 시:
   a. https://raw.githubusercontent.com/{USER}/{REPO}/main/{파일명} fetch
   b. frontmatter의 variables 목록 확인
   c. 내 메시지에서 각 변수값 추출
   d. 미확인 변수는 질문으로 수집
   e. {{variable}} 치환 후 실행

3. keyword 미감지: 일반 대화 응답

// PromptPipeline.js — 핵심 파이프라인 클래스

class PromptPipeline {
  constructor(config) {
    this.ghUser   = config.githubUser;
    this.ghRepo   = config.githubRepo;
    this.ghBranch = config.branch || 'main';
    this.ghToken  = config.token || null;
    this.aiKey    = config.aiKey;
    this.aiModel  = config.aiModel || 'gemini'; // 'gemini' | 'openai'
    this._index   = null; // 캐시
  }

  /* ① _INDEX.md 로드 및 파싱 (앱 시작 시 1회) */
  async loadIndex() {
    if (this._index) return this._index;
    const raw = await this._fetch('_INDEX.md');
    this._index = this._parseIndex(raw);
    return this._index;
  }

  _parseIndex(raw) {
    const map = {};
    const rows = raw.match(/\|[^|]+\|[^|]+\|[^|]+\|/g) || [];
    for (const row of rows) {
      const cols = row.split('|').map(s => s.trim()).filter(Boolean);
      if (cols.length < 2 || cols[0].startsWith('-')) continue;
      const keywords = cols[0].split(',').map(k => k.trim());
      const file     = cols[1].trim();
      const vars     = (cols[2] || '').split(',').map(v => v.trim());
      for (const kw of keywords) {
        if (kw && file) map[kw] = { file, vars };
      }
    }
    return map;
  }

  /* ② keyword 감지 */
  detectKeyword(userInput) {
    if (!this._index) throw new Error('Index not loaded');
    const lower = userInput.toLowerCase();
    for (const [kw, meta] of Object.entries(this._index)) {
      if (lower.includes(kw.toLowerCase())) {
        return { keyword: kw, ...meta };
      }
    }
    return null;
  }

  /* ③ GitHub raw fetch */
  async fetchTemplate(filename) {
    return await this._fetch(filename);
  }

  async _fetch(filename) {
    const url = `https://raw.githubusercontent.com/${this.ghUser}/${this.ghRepo}/${this.ghBranch}/${filename}`;
    const headers = { 'Accept': 'text/plain' };
    if (this.ghToken) headers['Authorization'] = `Bearer ${this.ghToken}`;
    const res = await fetch(url, { headers });
    if (!res.ok) throw new Error(`GitHub fetch 실패: ${res.status}`);
    return await res.text();
  }

  /* ④ 변수 추출 */
  extractVars(userInput, varList) {
    const extracted = {};
    const patterns = {
      location : /([가-힣a-zA-Z]{2,10})\s*(에서|로|근처|에|이|가)?/,
      companion: /(혼자|솔로|커플|가족|아기동반|친구|N명)/,
      budget   : /(\d+[\d,]*)\s*(만원|원|달러)/,
      season   : /(봄|여름|가을|겨울|봄철|여름철|\d+월)/,
      mood     : /(분위기|데이트|혼밥|회식|가족|조용한|활기찬)/,
      ticker   : /([A-Z]{1,5}|[가-힣]{2,6}(?:전자|화학|바이오)?)/,
      period   : /(단기|중기|장기|\d+개월)/,
      strategy : /(성장주|배당|모멘텀|가치투자)/,
    };
    for (const v of varList) {
      if (v && patterns[v]) {
        const m = userInput.match(patterns[v]);
        if (m) extracted[v] = m[1];
      }
    }
    return extracted;
  }

  /* ④-b 변수 치환 */
  substituteVars(template, vars) {
    let result = template.replace(/^---[\s\S]*?---\n/m, ''); // frontmatter 제거
    for (const [k, v] of Object.entries(vars)) {
      result = result.replace(new RegExp(`{{${k}}}`, 'g'), v || `[${k} 미입력]`);
    }
    return result.trim();
  }

  /* ⑤ AI API 호출 */
  async queryAI(finalPrompt) {
    if (this.aiModel === 'gemini') {
      return await this._callGemini(finalPrompt);
    } else {
      return await this._callOpenAI(finalPrompt);
    }
  }

  async _callGemini(prompt) {
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${this.aiKey}`;
    const body = { contents: [{ parts: [{ text: prompt }] }] };
    const res  = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) });
    const data = await res.json();
    return data.candidates?.[0]?.content?.parts?.[0]?.text || '응답 없음';
  }

  async _callOpenAI(prompt) {
    const res  = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${this.aiKey}` },
      body: JSON.stringify({ model: 'gpt-4o', messages: [{ role: 'user', content: prompt }] })
    });
    const data = await res.json();
    return data.choices?.[0]?.message?.content || '응답 없음';
  }

  /* ★ 전체 파이프라인 실행 */
  async run(userInput) {
    await this.loadIndex();
    const match = this.detectKeyword(userInput);
    if (!match) return { type: 'normal', prompt: userInput };

    const template    = await this.fetchTemplate(match.file);
    const vars        = this.extractVars(userInput, match.vars);
    const finalPrompt = this.substituteVars(template, vars);
    const response    = await this.queryAI(finalPrompt);
    return { type: 'template', keyword: match.keyword, file: match.file, vars, finalPrompt, response };
  }
}

// 사용 예시
const pipeline = new PromptPipeline({
  githubUser: 'your-id',
  githubRepo: 'prompt-templates',
  aiKey     : 'YOUR_API_KEY',
  aiModel   : 'gemini'
});

const result = await pipeline.run('강릉 1박2일 여행 커플로 봄에 10만원');

@Service
public class PromptPipelineService {

    private final RestTemplate restTemplate;
    private final Map<String, TemplateIndex> indexCache = new ConcurrentHashMap<>();

    private static final String GITHUB_RAW =
        "https://raw.githubusercontent.com/{user}/{repo}/{branch}/{file}";

    // ① 인덱스 로드 (앱 시작 시 @PostConstruct)
    @PostConstruct
    public void loadIndex() {
        String raw = fetchRaw("_INDEX.md");
        parseAndCacheIndex(raw);
    }

    // ② keyword 감지
    public TemplateIndex detectKeyword(String userInput) {
        String lower = userInput.toLowerCase();
        return indexCache.entrySet().stream()
            .filter(e -> lower.contains(e.getKey()))
            .map(Map.Entry::getValue)
            .findFirst()
            .orElse(null);
    }

    // ③ GitHub raw fetch
    public String fetchRaw(String filename) {
        String url = GITHUB_RAW
            .replace("{user}", githubUser)
            .replace("{repo}", githubRepo)
            .replace("{branch}", "main")
            .replace("{file}", filename);

        HttpHeaders headers = new HttpHeaders();
        if (StringUtils.hasText(githubToken)) {
            headers.set("Authorization", "Bearer " + githubToken);
        }
        return restTemplate.exchange(url, HttpMethod.GET,
            new HttpEntity<>(headers), String.class).getBody();
    }

    // ④ 변수 치환
    public String substituteVars(String template, Map<String, String> vars) {
        String result = removeFrontmatter(template);
        for (Map.Entry<String, String> e : vars.entrySet()) {

        return result;
